# app.py
echo "Creando archivo Flask app.py..."
cat > app.py <<'EOL'
import os
import json
import pymysql
import boto3
from flask import Flask, request, jsonify, render_template
from dotenv import load_dotenv

load_dotenv()

app = Flask(__name__)

RDS_ENDPOINT = os.environ['RDS_ENDPOINT']
RDS_USER = os.environ['RDS_USER']
RDS_PASSWORD = os.environ['RDS_PASSWORD']
RDS_DB = os.environ['RDS_DB']
SNS_TOPIC_ARN = os.environ['SNS_TOPIC_ARN']

# Conexión a RDS (puede fallar si no está en la red VPC)
try:
    conn = pymysql.connect(host=RDS_ENDPOINT, user=RDS_USER, password=RDS_PASSWORD, db=RDS_DB)
except Exception as e:
    print("No se pudo conectar a RDS:", e)
    conn = None

sns = boto3.client('sns')

if conn:
    with conn.cursor() as cur:
        cur.execute("""
            CREATE TABLE IF NOT EXISTS formulario (
                id INT AUTO_INCREMENT PRIMARY KEY,
                nombre VARCHAR(255) NOT NULL,
                email VARCHAR(255) NOT NULL,
                departamento ENUM('IT','RRHH','Comercial','Logística','Marketing') NOT NULL,
                asunto VARCHAR(255) NOT NULL,
                descripcion TEXT NOT NULL,
                fecha_registro TIMESTAMP DEFAULT CURRENT_TIMESTAMP
            )
        """)
        conn.commit()

@app.route('/')
def index():
    return render_template('index.html')

@app.route('/submit', methods=['POST'])
def submit_form():
    data = request.json
    required_fields = ['nombre','email','departamento','asunto','descripcion']
    for field in required_fields:
        if field not in data:
            return jsonify({"error": f"Falta el campo {field}"}), 400

    if conn:
        with conn.cursor() as cur:
            sql = """
                INSERT INTO formulario (nombre,email,departamento,asunto,descripcion)
                VALUES (%s,%s,%s,%s,%s)
            """
            cur.execute(sql, (data['nombre'], data['email'], data['departamento'], data['asunto'], data['descripcion']))
            conn.commit()
    sns.publish(TopicArn=SNS_TOPIC_ARN, Message=json.dumps(data))
    return jsonify({"message": "Datos recibidos y notificación enviada"}), 200

if __name__ == '__main__':
    app.run(host='0.0.0.0', port=8000)
EOL

# -----------------------------
# Configurar Gunicorn como servicio systemd
# -----------------------------
echo "Creando servicio systemd para Gunicorn..."
SERVICE_FILE=/etc/systemd/system/penguin.service
sudo bash -c "cat > $SERVICE_FILE" <<EOL
[Unit]
Description=Penguin On The Way Flask App
After=network.target

[Service]
User=ubuntu
WorkingDirectory=$APP_DIR
EnvironmentFile=/home/ubuntu/penguin_app/.env
ExecStart=$APP_DIR/venv/bin/gunicorn -w 4 -b 127.0.0.1:8000 app:app
Restart=always

[Install]
WantedBy=multi-user.target
EOL

sudo systemctl daemon-reload
sudo systemctl enable penguin.service
sudo systemctl start penguin.service

# -----------------------------
# Configurar Nginx
# -----------------------------
echo "Configurando Nginx como reverse proxy..."
NGINX_FILE=/etc/nginx/sites-available/penguin
sudo bash -c "cat > $NGINX_FILE" <<EOL
server {
    listen 80;
    server_name _;

    location / {
        proxy_pass http://127.0.0.1:8000;
        proxy_set_header Host \$host;
        proxy_set_header X-Real-IP \$remote_addr;
        proxy_set_header X-Forwarded-For \$proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto \$scheme;
    }
}
EOL

sudo ln -sf /etc/nginx/sites-available/penguin /etc/nginx/sites-enabled/penguin
sudo rm -f /etc/nginx/sites-enabled/default
sudo nginx -t
sudo systemctl restart nginx

echo "----------------------------------"
echo "Setup completo!"
echo "Penguin on the way estará disponible en http://<IP-de-ALB>/"
echo "Gunicorn arranca automáticamente al iniciar la máquina."
